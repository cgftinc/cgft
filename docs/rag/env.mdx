# creating the search environment

an environment defines the tools your model can use and the reward signals for training. for search/rag, this means a BM25 search tool and a reward function that checks if the model retrieved the right chunks.

## how it works

**the search tool** (`search_corpus`)
- gives the model BM25 search over your corpus
- takes a query string + optional metadata/filename filters
- returns ranked chunks with scores
- the model learns to write better queries + iteratively search for missing information via rl

**the reward function** (`compute_reward`)
- runs after the model generates an answer
- checks if the model's tool calls retrieved the reference chunks
- measures text overlap between retrieved content and ground truth chunks
- rewards ≥ 25% overlap, otherwise 0
- this teaches the model to search effectively, not just answer

during training, the model gets rewarded for using the search tool to find the right information before answering. over time it learns corpus-specific search patterns.

see [notebooks/synthetic_rag_dataset_gen.ipynb](https://github.com/cgft-io/synthetic-data-prep-lib/blob/main/notebooks/synthetic_rag_dataset_gen.ipynb) for full working example.

## searchenv implementation

```python
from benchmax.envs.base_env import BaseEnv, ToolDefinition
import aiohttp

class SearchEnv(BaseEnv):
    system_prompt = """Use the search tool to find relevant information.
Write your answer on the final line as <answer>your answer</answer>."""

    def __init__(self, api_key: str, corpus_id: str, base_url: str, dataset_path: str):
        self._api_key = api_key
        self._corpus_id = corpus_id
        self._base_url = base_url.rstrip("/")
        self._dataset_path = dataset_path

        # Define search tool
        search_tool = ToolDefinition(
            name="search_corpus",
            description="Search corpus using BM25",
            input_schema={
                "type": "object",
                "properties": {
                    "query": {"type": "string"},
                    "metadata": {"type": "object"},
                    "filename": {"type": "string"},
                    "limit": {"type": "integer"},
                },
                "required": ["query"],
            },
        )
        self._tools = {search_tool.name: (search_tool, self._search_corpus_tool)}

    async def _search_corpus_tool(
        self, query: str, metadata: dict = None, filename: str = None, limit: int = 10, **kwargs
    ) -> str:
        """Execute BM25 search against corpus API"""
        url = f"{self._base_url}/api/corpora/{self._corpus_id}/search"
        headers = {"Authorization": f"Bearer {self._api_key}"}

        async with aiohttp.ClientSession() as session:
            resp = await session.post(
                url,
                json={"query": query, "limit": limit, "metadata": metadata, "filename": filename},
                headers=headers,
            )
            data = await resp.json()

        # Format results for model
        results = data.get("results", [])
        lines = []
        for i, item in enumerate(results, 1):
            lines.append(f"{i}. {item['filename']} (score: {item['score']:.2f})")
            lines.append(f"   {item['content']}")
        return "\n".join(lines)

    async def compute_reward(self, rollout_id: str, completion: str, ground_truth: str, **kwargs):
        """Reward based on % of reference chunks found in completion"""
        reference_chunks = kwargs.get("reference_chunks", [])
        reference_text = " ".join([chunk["content"] for chunk in reference_chunks])

        # Simple text overlap check
        if reference_text:
            overlap = sum(c in completion for c in reference_text) / len(reference_text)
            return {"chunk_overlap": overlap if overlap >= 0.25 else 0.0}
        return {"chunk_overlap": 0.0}

    def get_train_val_split(self, train_ratio=0.7, seed=42):
        """Load dataset and stratify by qa_type"""
        from datasets import load_dataset, concatenate_datasets

        ds = load_dataset("json", data_files=self._dataset_path)["train"]
        qa_types = set(ds["qa_type"])

        train_splits, val_splits = [], []
        for qa_type in qa_types:
            type_subset = ds.filter(lambda x: x["qa_type"] == qa_type).shuffle(seed=seed)
            split_idx = int(len(type_subset) * train_ratio)
            train_splits.append(type_subset.select(range(split_idx)))
            val_splits.append(type_subset.select(range(split_idx, len(type_subset))))

        return concatenate_datasets(train_splits).shuffle(seed), concatenate_datasets(val_splits).shuffle(seed)

    # Required interface methods
    async def list_tools(self):
        return [self._tools[k][0] for k in sorted(self._tools)]

    async def run_tool(self, rollout_id: str, tool_name: str, **tool_args):
        _, tool_fn = self._tools[tool_name]
        return await tool_fn(**tool_args)

    @classmethod
    def dataset_preprocess(cls, example, **kwargs):
        from benchmax.envs.types import StandardizedExample
        return StandardizedExample(
            prompt=example["question"],
            ground_truth=example.get("answer"),
            init_rollout_args={},
        )

    async def init_rollout(self, rollout_id: str, **kwargs): pass
    async def release_rollout(self, rollout_id: str): pass
    async def shutdown(self): pass
```
